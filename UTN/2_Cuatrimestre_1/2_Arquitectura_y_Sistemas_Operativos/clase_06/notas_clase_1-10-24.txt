para montar particiones: sudo echo "/dev/sdc5	/datos/pelis	ext4	defaults 0 0" | sudo tee -a /etc/fstab
para montar particiones: sudo echo "/dev/sdc6	/datos/musica	ext4	defaults 0 0" | sudo tee -a /etc/fstab
para desmontar: sudo umount /datos/pelis /datos/musica
